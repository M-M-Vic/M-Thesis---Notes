\section{Experiment: models with a single active customer}
\label{sec:experiment}

The standard departure mechanisms studied so far --- abandonment at rate
$\theta_1 n_1$ (Model~$C_2$) and jockeying at rate $\gamma_1 n_1$
(Model~$B_2$) --- produce \emph{linear} departure rates proportional to the
queue length.  In the PGF equation this gives rise to derivative terms
$xy(\cdot)\partial P/\partial x$, making the equation a first-order ODE and
requiring the integrating-factor machinery of Sections~7--8.

This section explores a simpler variant: only the \emph{first customer in
line} (the head of Queue~1) can abandon or jockey, at a fixed rate
independent of queue length.  The aggregate departure rate is then
$\theta_1\cdot\mathbf{1}_{\{n_1\ge1\}}$ (not $\theta_1 n_1$), and the
corresponding PGF contribution is \emph{algebraic} rather than differential.
The resulting fundamental equation can be solved directly by the kernel
method, giving rational closed-form expressions for $P(x,y)$.

\paragraph*{Models $B_2^1$ and C$_2^1$.}
\begin{itemize}
    \item \textbf{Model~$B_2^1$:} $\gamma_2=\theta_1=\theta_2=0$, $\gamma_1>0$.
    Only the customer at the head of Queue~1 (the one that would be served
    next) may jockey to Queue~2, at rate $\gamma_1$.
    All other waiting Class-1 customers remain in Queue~1.
    \item \textbf{Model~$C_2^1$:} $\gamma_1=\gamma_2=\theta_2=0$, $\theta_1>0$.
    Only the customer at the head of Queue~1 may abandon the system,
    at rate $\theta_1$.  All other waiting Class-1 customers are patient.
\end{itemize}
In both models the service discipline, arrival processes, and Class-2
dynamics are identical to Models~B$_2$ and~C$_2$ respectively.

% ─────────────────────────────────────────────────────────────────────────────
\subsection{Balance equations}

\subsubsection*{Model~$B_2^1$}

The jockeying transition $(n_1,n_2)\to(n_1-1,n_2+1)$ occurs at the fixed
rate $\gamma_1$ whenever $n_1\ge1$.  The balance equations on $S$ are:
\begin{align}
    \begin{split}
        &\left[\lambda_1+\lambda_2+\mu+\gamma_1\right]\pi(n_1,n_2) \\
        &\quad=\quad \mu\,\pi(n_1+1,n_2)
          +\lambda_1\pi(n_1-1,n_2)
          +\lambda_2\pi(n_1,n_2-1)
          +\gamma_1\pi(n_1+1,n_2-1)
    \end{split}
    &&n_1\ge1,\;n_2\ge1, \label{eq:B21:interior} \\
    \begin{split}
        &\left[\lambda_1+\lambda_2+\mu+\gamma_1\right]\pi(n_1,0) \\
        &\quad=\quad \mu\,\pi(n_1+1,0)
          +\lambda_1\pi(n_1-1,0)
    \end{split}
    &&n_1\ge1,\;n_2=0, \label{eq:B21:xboundary} \\
    \begin{split}
        &\left[\lambda_1+\lambda_2+\mu\right]\pi(0,n_2) \\
        &\quad=\quad \mu\,\pi(1,n_2)
          +\gamma_1\pi(1,n_2-1)
          +\lambda_2\pi(0,n_2-1)
          +\mu\,\pi(0,n_2+1)
    \end{split}
    &&n_1=0,\;n_2\ge1, \label{eq:B21:yboundary} \\
    \begin{split}
        &\left[\lambda_1+\lambda_2+\mu\right]\pi(0,0)
          = (\lambda_1+\lambda_2)\pi_0
          + \mu\,\pi(1,0)
          + \mu\,\pi(0,1),
    \end{split} &&\label{eq:B21:zero} \\
    \begin{split}
        &(\lambda_1+\lambda_2)\pi_0 = \mu\,\pi(0,0).
    \end{split} &&\label{eq:B21:idle}
\end{align}
The key structural difference from Model~$B_2$ (Equation~\eqref{eq:gen:balance_interior}
with $\gamma_2=\theta_i=0$) is that the jockeying coefficient is $\gamma_1$
(constant) rather than $\gamma_1 n_1$ (linear in queue length).

\subsubsection*{Model~$C_2^1$}

The abandonment transition $(n_1,n_2)\to(n_1-1,n_2)$ occurs at the fixed
rate $\theta_1$ whenever $n_1\ge1$.  The balance equations are:
\begin{align}
    \begin{split}
        &\left[\lambda_1+\lambda_2+\mu+\theta_1\right]\pi(n_1,n_2) \\
        &\quad=\quad (\mu+\theta_1)\pi(n_1+1,n_2)
          +\lambda_1\pi(n_1-1,n_2)
          +\lambda_2\pi(n_1,n_2-1)
    \end{split}
    &&n_1\ge1,\;n_2\ge1, \label{eq:C21:interior} \\
    \begin{split}
        &\left[\lambda_1+\lambda_2+\mu+\theta_1\right]\pi(n_1,0) \\
        &\quad=\quad (\mu+\theta_1)\pi(n_1+1,0)
          +\lambda_1\pi(n_1-1,0)
    \end{split}
    &&n_1\ge1,\;n_2=0, \label{eq:C21:xboundary} \\
    \begin{split}
        &\left[\lambda_1+\lambda_2+\mu\right]\pi(0,n_2) \\
        &\quad=\quad (\mu+\theta_1)\pi(1,n_2)
          +\lambda_2\pi(0,n_2-1)
          +\mu\,\pi(0,n_2+1)
    \end{split}
    &&n_1=0,\;n_2\ge1, \label{eq:C21:yboundary} \\
    \begin{split}
        &\left[\lambda_1+\lambda_2+\mu\right]\pi(0,0)
          = (\lambda_1+\lambda_2)\pi_0
          + (\mu+\theta_1)\pi(1,0)
          + \mu\,\pi(0,1),
    \end{split} &&\label{eq:C21:zero} \\
    \begin{split}
        &(\lambda_1+\lambda_2)\pi_0 = \mu\,\pi(0,0).
    \end{split} &&\label{eq:C21:idle}
\end{align}
In Model~$C_2$, the coefficients $\mu+\theta_1(n_1+1)$ and $\theta_1 n_1$
grow with $n_1$.  Here they are constants $\mu+\theta_1$ and $\theta_1$
throughout; this is the departure point of the analysis.

% ─────────────────────────────────────────────────────────────────────────────
\subsection{Fundamental PGF equations}

\begin{lemma}[Fundamental equation for Model~$B_2^1$]\label{lem:B21:fundamental}
The joint PGF $P(x,y)$ satisfies
\begin{equation}
    \begin{split}
        &\left[(\lambda_1+\lambda_2+\mu)xy - \mu y - \lambda_1 x^2y - \lambda_2 xy^2
               + \gamma_1 y(x-y)\right]P(x,y) \\
        &\quad=\quad \left[\mu(x-y) + \gamma_1 y(x-y)\right]P_y(y)
               - \mu x(1-y)\,\pi(0,0).
    \end{split}
    \label{eq:B21:fundamental}
\end{equation}
\end{lemma}

\begin{proof}
Multiply the interior equation~\eqref{eq:B21:interior} by $x^{n_1}y^{n_2}$
and sum over $n_1\ge1$, $n_2\ge1$.  The jockeying term generates:
\begin{align}
    \begin{split}
        \text{Rate out: } &\gamma_1\sum_{n_1\ge1,\,n_2\ge1}\pi(n_1,n_2)\,x^{n_1}y^{n_2}
        = \gamma_1\bigl[P-P_y\bigr], \\
        \text{Rate in: } &\gamma_1\sum_{n_1\ge1,\,n_2\ge1}\pi(n_1+1,n_2-1)\,x^{n_1}y^{n_2}
        = \gamma_1\,\tfrac{y}{x}\bigl[P-P_y\bigr]+\text{(boundary corrections)}.
    \end{split}
\end{align}
The net contribution is $\gamma_1(1-y/x)(P-P_y)$.
After repeating the procedure for the
$x$-boundary~\eqref{eq:B21:xboundary}, the $y$-boundary~\eqref{eq:B21:yboundary},
and state~\eqref{eq:B21:zero}, the boundary corrections cancel exactly
(same algebra as in Lemma~\ref{lem:gen:fundamental_equation}) and multiplying
through by $xy$ yields~\eqref{eq:B21:fundamental}.
\end{proof}

\textit{Remark.}
Equation~\eqref{eq:B21:fundamental} contains \emph{no derivative} of $P$.
Contrast with Model~$B_2$, whose fundamental equation has the term
$\gamma_1 xy(x-y)\partial P/\partial x$.
The flat jockeying rate contributes the algebraic factor $\gamma_1y(x-y)$ to
the coefficient of $P(x,y)$, reducing the functional equation to an algebraic one.

\begin{lemma}[Fundamental equation for Model~$C_2^1$]\label{lem:C21:fundamental}
The joint PGF $P(x,y)$ satisfies
\begin{equation}
    \begin{split}
        &\left[(\lambda_1+\lambda_2+\mu)xy - \mu y - \lambda_1 x^2y - \lambda_2 xy^2
               + \theta_1 y(x-1)\right]P(x,y) \\
        &\quad=\quad \left[\mu(x-y) + \theta_1 y(x-1)\right]P_y(y)
               - \mu x(1-y)\,\pi(0,0).
    \end{split}
    \label{eq:C21:fundamental}
\end{equation}
\end{lemma}

\begin{proof}
The abandonment term in~\eqref{eq:C21:interior} contributes:
\begin{align}
    \begin{split}
        \text{Rate out: }
        &\theta_1\sum_{n_1\ge1,\,n_2\ge1}\pi(n_1,n_2)\,x^{n_1}y^{n_2}
         = \theta_1\bigl[P-P_y\bigr], \\
        \text{Rate in: }
        &\theta_1\sum_{n_1\ge1,\,n_2\ge1}\pi(n_1+1,n_2)\,x^{n_1}y^{n_2}
         = \theta_1 x^{-1}\bigl[P-P_y\bigr]+\text{(boundary corrections)}.
    \end{split}
\end{align}
The net contribution is $\theta_1(1-x^{-1})(P-P_y)$.
Combining all boundary equations as before and multiplying by $xy$
yields~\eqref{eq:C21:fundamental}.
\end{proof}

\textit{Remark.}
Equation~\eqref{eq:C21:fundamental} likewise contains no derivative of $P$.
The term replacing $\theta_1 xy(x-1)\partial P/\partial x$ of Model~$C_2$ is
the algebraic factor $\theta_1 y(x-1)$ in the coefficient of $P(x,y)$.

% ─────────────────────────────────────────────────────────────────────────────
\subsection{Closed-form solution via the kernel method}

Both fundamental equations have the structure
\begin{equation}
    D(x,y)\,P(x,y) \;=\; R(x,y)\,P_y(y) - \mu x(1-y)\,\pi(0,0),
    \label{eq:exp:algebraic}
\end{equation}
where $D(x,y)$ is polynomial and $R(x,y)$ does not involve $P$.
Since there is no derivative of $P$, this yields directly
$P(x,y) = [R\,P_y - \mu x(1-y)\pi(0,0)]/D$
once $P_y(y)$ and $\pi(0,0)$ are known.

\subsubsection*{Step 1 — characteristic roots}

For each model, the factor $y$ in $D$ cancels and the inner quadratic in $x$ is:
\begin{align}
    \begin{split}
        &\text{B$_2^1$:}\quad
        \lambda_1 x^2 - \bigl(\lambda_1+\lambda_2(1-y)+\mu+\gamma_1\bigr)x + (\mu+\gamma_1 y) = 0,
    \end{split}
    \label{eq:B21:quad} \\
    \begin{split}
        &\text{C$_2^1$:}\quad
        \lambda_1 x^2 - \bigl(\lambda_1+\lambda_2(1-y)+\mu+\theta_1\bigr)x + (\mu+\theta_1) = 0.
    \end{split}
    \label{eq:C21:quad}
\end{align}
Both have the same form as Model~$A$'s quadratic~\eqref{eq:A:xstar}, but with
$\mu$ replaced by $\mu+\gamma_1 y$ (B$_2^1$) or $\mu+\theta_1$ (C$_2^1$).
By Vieta's formulas, the product of the two roots is
$(\mu+\gamma_1 y)/\lambda_1$ or $(\mu+\theta_1)/\lambda_1$ respectively.
The negative-sign (inner) roots are:
\begin{align}
    \begin{split}
        x_{B_2^1}^*(y) &= \frac{A_{B}(y) - \sqrt{A_{B}(y)^2 - 4\lambda_1(\mu+\gamma_1 y)}}
        {2\lambda_1}, \quad A_{B}(y)=\lambda_1+\lambda_2(1-y)+\mu+\gamma_1,
    \end{split}
    \label{eq:B21:xstar} \\
    \begin{split}
        x_{C_2^1}^*(y) &= \frac{A_{C} - \sqrt{A_{C}^2 - 4\lambda_1(\mu+\theta_1)}}
        {2\lambda_1}, \quad A_{C}=\lambda_1+\lambda_2(1-y)+\mu+\theta_1.
    \end{split}
    \label{eq:C21:xstar}
\end{align}
At $y=1$: $x_{B_2^1}^*(1)=1$ and $x_{C_2^1}^*(1)=1$ (both verified by direct
substitution).  At $\gamma_1=0$ (resp.\ $\theta_1=0$) both roots reduce to
$x^*(y)$ of Model~$A$~\eqref{eq:A:xstar}.

\subsubsection*{Step 2 — boundary function $P_y(y)$}

At $x=x_r^*(y)$ the denominator $D$ vanishes.  For $P$ to be bounded, the
numerator must also vanish.  Solving for $P_y(y)$:
\begin{align}
    \begin{split}
        P_y^{B_2^1}(y) \;&=\;
        \frac{\mu\, x_{B_2^1}^*(y)\,(1-y)\,\pi(0,0)}
        {\bigl(x_{B_2^1}^*(y)-y\bigr)\,(\mu+\gamma_1 y)},
    \end{split}
    \label{eq:B21:Py} \\[4pt]
    \begin{split}
        P_y^{C_2^1}(y) \;&=\;
        \frac{\mu\, x_{C_2^1}^*(y)\,(1-y)\,\pi(0,0)}
        {(\mu+\theta_1 y)\,x_{C_2^1}^*(y) - y(\mu+\theta_1)}.
    \end{split}
    \label{eq:C21:Py}
\end{align}
Setting $y=0$: using $x_r^*(0)^2 \cdot \lambda_1 = \mu + (\text{mechanism at }y=0)$
and Vieta's product, one verifies $P_y(0)=\pi(0,0)$ in both cases, confirming
the boundary condition.

\subsubsection*{Step 3 — joint PGF}

Substituting~\eqref{eq:B21:Py} (resp.\ \eqref{eq:C21:Py})
into~\eqref{eq:exp:algebraic} and simplifying, the factor $(x-x_r^*)$
cancels between numerator and denominator.
Denoting $x_r^* = x_{B_2^1}^*$ or $x_{C_2^1}^*$, and
$x_r^+$ for the corresponding outer root:
\begin{align}
    \begin{split}
        P_{B_2^1}(x,y) \;&=\;
        \frac{\mu\,\rho(1-\rho)\,(1-y)}
        {\lambda_1\,\bigl(x_{B_2^1}^*(y)-y\bigr)\,\bigl(x_{B_2^1}^+(y)-x\bigr)},
    \end{split}
    \label{eq:B21:Pxy} \\[8pt]
    \begin{split}
        P_{C_2^1}(x,y) \;&=\;
        \frac{\mu(\mu+\theta_1)\,(1-y)\,\pi(0,0)}
        {\lambda_1\,\bigl(x_{C_2^1}^+(y)-x\bigr)
         \,\bigl[(\mu+\theta_1 y)\,x_{C_2^1}^*(y)-y(\mu+\theta_1)\bigr]}.
    \end{split}
    \label{eq:C21:Pxy}
\end{align}

\textit{Remark.}
Formula~\eqref{eq:B21:Pxy} is \emph{structurally identical to Model~$A$}:
\begin{equation}
    P_A(x,y) \;=\; \frac{\mu\,\rho(1-\rho)\,(1-y)}
    {\lambda_1\,(x_A^*(y)-y)\,(x_A^+(y)-x)}.
    \label{eq:A:Pxy_factored}
\end{equation}
The only difference between Models $A$ and B$_2^1$ is in the characteristic
roots: replacing $\mu\to\mu+\gamma_1 y$ in the quadratic shifts the roots
from $x_A^*,x_A^+$ to $x_{B_2^1}^*,x_{B_2^1}^+$.

% ─────────────────────────────────────────────────────────────────────────────
\subsection{Empty-state probabilities}

\subsubsection*{Model~$B_2^1$}

Jockeying conserves the total number of customers, so the same diagonal
argument as in Model~$A$ (Equation~\eqref{eq:A:Pzz}) gives
$P(z,z)=\rho(1-\rho)/(1-\rho z)$, hence
\begin{equation}
    \pi(0,0) \;=\; \rho(1-\rho), \qquad \pi_0 \;=\; 1-\rho. \label{eq:B21:pi00}
\end{equation}

\subsubsection*{Model~$C_2^1$}

Abandonment reduces the load, so conservation breaks and a different argument
is needed.  Setting $y=1$ in~\eqref{eq:C21:fundamental}: the factor
$\mu x(1-y)$ on the right vanishes, and $R^{C_2^1}(x,1)=(\mu+\theta_1)(x-1)$.
The left-hand polynomial in $x$ factors as
$-\lambda_1(x-1)(x-(\mu+\theta_1)/\lambda_1)$.
Cancelling $(x-1)$:
\begin{align}
    \begin{split}
        P(x,1) \;=\; \frac{P_y(1)}{1-\rho_C\,x}, \qquad
        \rho_C \;\equiv\; \frac{\lambda_1}{\mu+\theta_1}.
    \end{split}
    \label{eq:C21:Px1}
\end{align}
This is a geometric marginal: Class-1's effective ``service rate'' is
$\mu+\theta_1$ (completions plus first-customer abandonments).
Evaluating at $x=1$:
\begin{align}
    \begin{split}
        P(1,1) \;=\; \frac{P_y(1)}{1-\rho_C} \;=\; 1-\pi_0. \label{eq:C21:P11}
    \end{split}
\end{align}
A second relation comes from setting $x=1$ in~\eqref{eq:C21:fundamental}
and using $D^{C_2^1}(1,y)=\lambda_2 y(1-y)$:
\begin{align}
    \begin{split}
        \lambda_2\,P(1,y) \;=\; \frac{\mu}{y}\bigl[P_y(y)-\pi(0,0)\bigr].
    \end{split}
\end{align}
Taking $y\to1$ and using~\eqref{eq:C21:P11} and the idle
balance $\lambda\pi_0=\mu\pi(0,0)$:
\begin{align}
    \begin{split}
        \lambda_2(1-\pi_0) \;=\; \mu\bigl[P_y(1)-\pi(0,0)\bigr]
        \;=\; \mu\bigl[(1-\pi_0)(1-\rho_C) - \pi(0,0)\bigr].
    \end{split}
\end{align}
Substituting $\pi(0,0)=\lambda\pi_0/\mu$ and solving:
\begin{align}
    \begin{split}
        \pi_0 \;=\; \frac{(\mu+\theta_1)(\mu-\lambda)+\lambda_1\theta_1}
                        {\mu(\mu+\theta_1)+\lambda_1\theta_1},
        \qquad
        \pi(0,0) \;=\; \frac{\lambda\bigl[(\mu+\theta_1)(\mu-\lambda)+\lambda_1\theta_1\bigr]}
                             {\mu\bigl[\mu(\mu+\theta_1)+\lambda_1\theta_1\bigr]}.
    \end{split}
    \label{eq:C21:pi0_pi00}
\end{align}
At $\theta_1=0$: $\pi_0\to1-\rho$ and $\pi(0,0)\to\rho(1-\rho)$, recovering Model~$A$. \\
Stability requires $\pi_0>0$, i.e.\ $(\mu+\theta_1)(\mu-\lambda)+\lambda_1\theta_1>0$,
a \emph{weaker} condition than $\rho<1$.

% [Note: stability can hold even for λ > μ provided θ₁ is large enough —
%  abandonment from the head of Queue 1 accelerates the clearance of the
%  priority queue, stabilising the system at higher offered loads.]

% ─────────────────────────────────────────────────────────────────────────────
\subsection{Main results and comparison}

\begin{theorem}[Closed-form PGF for Models $B_2^1$ and C$_2^1$]\label{thm:exp}
Under stability, the joint PGF of the two-class non-preemptive priority queue
is rational in both experiment models.

\emph{(i) Model~$B_2^1$.}  With $\pi(0,0)=\rho(1-\rho)$,
\begin{equation}
    P_{B_2^1}(x,y) \;=\;
    \frac{\mu\,\rho(1-\rho)\,(1-y)}
    {\lambda_1\,\bigl(x_{B_2^1}^*(y)-y\bigr)\,\bigl(x_{B_2^1}^+(y)-x\bigr)},
    \label{eq:B21:theorem}
\end{equation}
where $x_{B_2^1}^*(y)$ and $x_{B_2^1}^+(y)$ are the two roots
of~\eqref{eq:B21:quad}.  This has the \emph{same rational form as Model~$A$},
differing only in the characteristic roots
($\mu\to\mu+\gamma_1 y$ in the quadratic).

\emph{(ii) Model~$C_2^1$.}  With $\pi(0,0)$ given by~\eqref{eq:C21:pi0_pi00},
\begin{equation}
    P_{C_2^1}(x,y) \;=\;
    \frac{\mu(\mu+\theta_1)\,(1-y)\,\pi(0,0)}
    {\lambda_1\,\bigl(x_{C_2^1}^+(y)-x\bigr)
     \,\bigl[(\mu+\theta_1 y)\,x_{C_2^1}^*(y)-y(\mu+\theta_1)\bigr]},
    \label{eq:C21:theorem}
\end{equation}
where $x_{C_2^1}^*(y)$ and $x_{C_2^1}^+(y)$ are the roots of~\eqref{eq:C21:quad}.
At $\theta_1\to0^+$ the formula recovers Model~$A$ exactly.
\end{theorem}

The ``first-customer-only'' assumption converts both ODEs into algebraic
equations.  For B$_2^1$, the result is structurally identical to Model~$A$
— the only change is $\mu\to\mu+\gamma_1 y$ in the quadratic that defines
the characteristic root.  For C$_2^1$, the structure is similar but the
extra factor $(\mu+\theta_1)$ in the numerator and the modified denominator
reflect the broken conservation.  In both cases all performance measures
follow by direct differentiation of a rational function, with no quadrature
required.
